# 二、代码

inline void read(int & a)
{
    char ch;
    int fh=1;
    while(!(isdigit(ch)))
    {
        if(ch=='-')
            fh=-1;
        ch=getchar();
    }
    a=ch-'0';
    ch=getchar();
    while(isdigit(ch))
        a=(a<<3)+(a<<1)+ch-'0',ch=getchar();
    a*=fh;
}
inline void write(int x)
{
    if(x<0)
	{//判断负数
        x=-x;
        putchar('-');
    }
    if(x>9) write(x/10);//位值拆数，递归
    putchar(x%10+'0');//按位输出
}

#include <bits/stdc++.h>

int main()
{
    int T=1000;
    while(T--)
    {
        system("data.exe >  1.in");
        system("my.exe <1.in > 1.out");
        system("bc.exe <1.in > 2.out");
        if(system("fc 1.out 2.out"))
            break;
    }
    system("pause");
    return 0;
}

my为自己的高效代码
